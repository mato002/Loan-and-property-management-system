        function loanBranchPicker(selectId, branchStoreUrl) {\n

                async saveBranch() {\n
                    this.branchModalError = '';\n

                        this.branchModalError = 'Branch name is required.';\n
                        return;\n
                    }\n
\n
                    this.isSavingBranch = true;\n
                    try {\n
                        const response = await fetch(this.branchStoreUrl, {\n
                            method: 'POST',\n
                            headers: {\n
                                'Accept': 'application/json',\n
                                'Content-Type': 'application/json',\n
                                'X-CSRF-TOKEN': document.querySelector('meta[name="csrf-token"]')?.getAttribute('content') ?? '',\n

                            body: JSON.stringify({ name: this.newBranchName }),\n

\n
                        const data = await response.json();\n
                        if (!response.ok || !data?.ok || !data?.branch?.name) {\n
                            this.branchModalError = data?.message ?? 'Unable to save branch.';\n
                            return;\n
                        }\n
\n
                        const branchSelect = document.getElementById(this.selectId);\n
                        if (!branchSelect) {\n
                            this.branchModalError = 'Unable to locate branch field.';\n
                            return;\n
